    ${CMAKE_CURRENT_SOURCE_DIR}
    ${CMAKE_CURRENT_SOURCE_DIR}/../
    ../../Externals/lzo/include
    ../../Externals/luabind
    ${SDL_INCLUDE_DIRS})
